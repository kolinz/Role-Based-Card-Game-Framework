```
async fetchConfig() {
    try {
        const res  = await fetch('/api/config');
        const data = await res.json();
        if (data.ok) {
            this._cardDisplayDuration      = data.cardDisplayDuration;
            this._cardDisplayDurationShort = data.cardDisplayDurationShort;

            this._diceRollDuration         = data.diceRollDuration  ?? 2.2;
            this._wsReconnectDelay         = data.wsReconnectDelay  ?? 3000;

    }
```

```
## handleServerMessage() の重要ルール（v1.3.1）

### render() を呼ばないケース
以下のメッセージ受信時は render() を呼ぶとボタン状態がリセットされるため禁止:
- diceRolled   → showCardArea() + setBtnState() のみ
- cardSelected  → _showCardResult() + upScoreboard() のみ ← render() 禁止
- cardSelectedByOther → sceneManager.showCardOnTable() + _showCardResult() + upScoreboard() のみ

### initScene() は必ず render() の後に呼ぶ
// ✅ 正しい順序
this.mode = 'lobby';
this.render();     // 先に #scene-container を DOM に生成
this.initScene();  // その後 SceneManager を起動

// 適用: confirmAvatar() / gameStarted / gameReset

### gameStarted ハンドラ
case 'gameStarted':
    this.session = data.session;
    this.mode    = 'game';
    this.render();        // 先に DOM 生成
    this.initScene();     // SceneManager 起動
    this.upScoreboard();
    break;

### cardSelectedByOther ハンドラ（v1.3.1 カード表示追加）
case 'cardSelectedByOther':
    this.session = data.session;
    if (this.sceneManager) {
        const p = this.session.players.find(pl => pl.id === data.playerId);
        if (p) this.sceneManager.showCardOnTable(p.id, data.cardType);
    }
    // 他プレイヤーが選んだカードの内容を表示（v1.3.1 追加）
    if (data.card) {
        const playerName = data.playerName ?? '';
        this._showCardResult(data.card, data.matched, false, true, playerName);
    }
    this.upScoreboard();
    break;

### cardSelected ハンドラ（v1.3.1 _showCardResult に統一）
case 'cardSelected':
    this.session = data.session;
    if (data.card) {
        this._showCardResult(data.card, data.matched, data.alreadySelected, false);
    }
    if (data.playerFinished) setTimeout(() => this.showGoalModal(data), 2400);
    if (data.allFinished) setTimeout(() => this.showFinalRankingsModal(data.finalRankings), 3500);
    this.upScoreboard();
    break;
```
